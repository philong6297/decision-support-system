add_custom_target(
  copy_data
  ALL
  COMMAND ${CMAKE_COMMAND} -E copy_directory "${LONGLP_PROJECT_DATA_DIR}" "${LONGLP_PROJECT_OUTPUT_DIR}/data"
  VERBATIM
  WORKING_DIRECTORY ${LONGLP_PROJECT_DIR}
  USES_TERMINAL
  COMMENT "Copying ${LONGLP_PROJECT_DATA_DIR} to ${LONGLP_PROJECT_OUTPUT_DIR}/data"
)
